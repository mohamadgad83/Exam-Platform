🗄️ هيكل قاعدة البيانات (Supabase)

-- ============================================ -- database.sql - جميع الجداول (INTEGER IDs) -- لمنصة الاختبارات المتكاملة -- ============================================ -- 1. جدول المستخدمين CREATE TABLE users ( id SERIAL PRIMARY KEY, name VARCHAR(255) NOT NULL, username VARCHAR(100) UNIQUE, phone VARCHAR(50) UNIQUE, email VARCHAR(255) UNIQUE, password_hash VARCHAR(255) NOT NULL, role VARCHAR(50) CHECK (role IN ('admin', 'teacher', 'student')), status VARCHAR(50) DEFAULT 'pending', class_id INTEGER, created_at TIMESTAMP DEFAULT NOW() ); -- 2. جدول الصفوف CREATE TABLE classes ( id SERIAL PRIMARY KEY, name VARCHAR(255) NOT NULL, grade_level VARCHAR(100), description TEXT, created_at TIMESTAMP DEFAULT NOW() ); -- 3. جدول المواد CREATE TABLE subjects ( id SERIAL PRIMARY KEY, name VARCHAR(255) NOT NULL, description TEXT, created_at TIMESTAMP DEFAULT NOW() ); -- 4. جدول ربط المواد بالصفوف CREATE TABLE class_subjects ( id SERIAL PRIMARY KEY, class_id INTEGER REFERENCES classes(id) ON DELETE CASCADE, subject_id INTEGER REFERENCES subjects(id) ON DELETE CASCADE, created_at TIMESTAMP DEFAULT NOW(), UNIQUE(class_id, subject_id) ); -- 5. جدول تعيينات المعلمين CREATE TABLE teacher_assignments ( id SERIAL PRIMARY KEY, teacher_id INTEGER REFERENCES users(id) ON DELETE CASCADE, class_id INTEGER REFERENCES classes(id) ON DELETE CASCADE, subject_id INTEGER REFERENCES subjects(id) ON DELETE CASCADE, created_at TIMESTAMP DEFAULT NOW() ); -- 6. جدول الامتحانات CREATE TABLE exams ( id SERIAL PRIMARY KEY, title VARCHAR(255) NOT NULL, description TEXT, class_id INTEGER REFERENCES classes(id), subject_id INTEGER REFERENCES subjects(id), teacher_id INTEGER REFERENCES users(id), duration_minutes INTEGER DEFAULT 60, passing_score INTEGER DEFAULT 50, exam_type VARCHAR(20) DEFAULT 'free', exam_price DECIMAL(10,2) DEFAULT 0, target_type VARCHAR(20) DEFAULT 'all', target_ids TEXT, is_announced BOOLEAN DEFAULT false, announced_at TIMESTAMP, start_date TIMESTAMP, end_date TIMESTAMP, status VARCHAR(50) DEFAULT 'draft', created_by INTEGER REFERENCES users(id), max_attempts INTEGER DEFAULT 1, created_at TIMESTAMP DEFAULT NOW() ); -- 7. جدول exam_questions (أسئلة الامتحان) CREATE TABLE exam_questions ( id SERIAL PRIMARY KEY, exam_id INTEGER REFERENCES exams(id) ON DELETE CASCADE, question_id INTEGER REFERENCES questions(id), points DECIMAL(5,2) DEFAULT 1, order_index INTEGER DEFAULT 0, created_at TIMESTAMP DEFAULT NOW() ); -- 8. جدول الأسئلة CREATE TABLE questions ( id SERIAL PRIMARY KEY, teacher_id INTEGER REFERENCES users(id), class_id INTEGER REFERENCES classes(id), subject_id INTEGER REFERENCES subjects(id), question_text TEXT, question_image TEXT, type VARCHAR(50) DEFAULT 'mcq', points INTEGER DEFAULT 5, difficulty VARCHAR(20) DEFAULT 'medium', options TEXT, correct_answer TEXT, model_answer TEXT, status VARCHAR(50) DEFAULT 'pending', feedback TEXT, created_at TIMESTAMP DEFAULT NOW() ); -- 9. جدول المجموعات CREATE TABLE groups ( id SERIAL PRIMARY KEY, name VARCHAR(255) NOT NULL, description TEXT, class_id INTEGER REFERENCES classes(id), subject_id INTEGER REFERENCES subjects(id), teacher_id INTEGER REFERENCES users(id), max_capacity INTEGER, is_open BOOLEAN DEFAULT true, created_at TIMESTAMP DEFAULT NOW() ); -- 10. جدول أعضاء المجموعات CREATE TABLE group_members ( id SERIAL PRIMARY KEY, group_id INTEGER REFERENCES groups(id) ON DELETE CASCADE, student_id INTEGER REFERENCES users(id) ON DELETE CASCADE, status VARCHAR(20) DEFAULT 'pending', joined_at TIMESTAMP, created_at TIMESTAMP DEFAULT NOW(), UNIQUE(group_id, student_id) ); -- 11. جدول exam_requests (طلبات الاشتراك) CREATE TABLE exam_requests ( id SERIAL PRIMARY KEY, exam_id INTEGER REFERENCES exams(id) ON DELETE CASCADE, student_id INTEGER REFERENCES users(id) ON DELETE CASCADE, status VARCHAR(20) DEFAULT 'pending_admin', approved_by INTEGER REFERENCES users(id), approved_at TIMESTAMP, rejection_reason TEXT, coupon_code VARCHAR(50), discount_amount DECIMAL(10,2) DEFAULT 0, final_amount DECIMAL(10,2), created_at TIMESTAMP DEFAULT NOW(), UNIQUE(exam_id, student_id) ); -- 12. جدول exam_enrollments (الطلاب المقبولون) CREATE TABLE exam_enrollments ( id SERIAL PRIMARY KEY, exam_id INTEGER REFERENCES exams(id) ON DELETE CASCADE, student_id INTEGER REFERENCES users(id) ON DELETE CASCADE, exam_request_id INTEGER REFERENCES exam_requests(id), enrolled_by INTEGER REFERENCES users(id), enrolled_at TIMESTAMP DEFAULT NOW(), UNIQUE(exam_id, student_id) ); -- 13. جدول exam_attempts (محاولات الامتحان) CREATE TABLE exam_attempts ( id SERIAL PRIMARY KEY, exam_id INTEGER REFERENCES exams(id), student_id INTEGER REFERENCES users(id), start_time TIMESTAMP, submitted_at TIMESTAMP, answers JSONB, essay_scores JSONB, score DECIMAL(10,2), total_points DECIMAL(10,2), status VARCHAR(20) DEFAULT 'in_progress', enrollment_id INTEGER REFERENCES exam_enrollments(id), created_at TIMESTAMP DEFAULT NOW() ); -- 14. جدول exam_violations (مخالفات الامتحان) CREATE TABLE exam_violations ( id SERIAL PRIMARY KEY, exam_attempt_id INTEGER REFERENCES exam_attempts(id), student_id INTEGER REFERENCES users(id), violation_type VARCHAR(100), details TEXT, timestamp TIMESTAMP DEFAULT NOW() ); -- 15. جدول payment_methods (طرق الدفع) CREATE TABLE payment_methods ( id SERIAL PRIMARY KEY, name VARCHAR(100) NOT NULL, description TEXT, account_number VARCHAR(100), account_owner VARCHAR(100), is_active BOOLEAN DEFAULT true, sort_order INTEGER DEFAULT 0, created_by INTEGER REFERENCES users(id), created_at TIMESTAMP DEFAULT NOW() ); -- 16. جدول payments (المدفوعات) CREATE TABLE payments ( id SERIAL PRIMARY KEY, exam_request_id INTEGER REFERENCES exam_requests(id) ON DELETE CASCADE, student_id INTEGER REFERENCES users(id), exam_id INTEGER REFERENCES exams(id), amount DECIMAL(10,2), payment_method_id INTEGER REFERENCES payment_methods(id), payment_screenshot TEXT, status VARCHAR(20) DEFAULT 'pending', confirmed_by INTEGER REFERENCES users(id), confirmed_at TIMESTAMP, rejection_reason TEXT, created_at TIMESTAMP DEFAULT NOW() ); -- 17. جدول coupons (كوبونات الخصم) CREATE TABLE coupons ( id SERIAL PRIMARY KEY, code VARCHAR(50) UNIQUE NOT NULL, discount_type VARCHAR(20) DEFAULT 'percentage', discount_value DECIMAL(10,2) NOT NULL, max_discount DECIMAL(10,2), min_amount DECIMAL(10,2), valid_from TIMESTAMP, valid_until TIMESTAMP, usage_limit INTEGER DEFAULT 1, used_count INTEGER DEFAULT 0, target_exam_id INTEGER REFERENCES exams(id), target_students TEXT, is_active BOOLEAN DEFAULT true, created_by INTEGER REFERENCES users(id), created_at TIMESTAMP DEFAULT NOW() ); -- 18. جدول coupon_usages (استخدامات الكوبونات) CREATE TABLE coupon_usages ( id SERIAL PRIMARY KEY, coupon_id INTEGER REFERENCES coupons(id), student_id INTEGER REFERENCES users(id), exam_request_id INTEGER REFERENCES exam_requests(id), used_at TIMESTAMP DEFAULT NOW() ); -- 19. جدول notifications (الإشعارات الداخلية) CREATE TABLE notifications ( id SERIAL PRIMARY KEY, user_id INTEGER REFERENCES users(id) ON DELETE CASCADE, title VARCHAR(255), message TEXT, type VARCHAR(50), related_id VARCHAR(50), is_read BOOLEAN DEFAULT false, created_at TIMESTAMP DEFAULT NOW() );

🔐 نظام تسجيل الدخول والصلاحيات

الدورطريقة الدخولالصلاحيات
أدمناسم مستخدم + كلمة سريرى كل شيء - صلاحيات كاملة
معلماسم مستخدم + كلمة سريرى فقط ما يخصه (صفوفه، مواده، طلابه)
طالبرقم هاتف + كلمة سريرى فقط الامتحانات المنشورة في صفه
