🗄️ هيكل قاعدة البيانات (Supabase)

-- جدول المستخدمين CREATE TABLE users ( id SERIAL PRIMARY KEY, name VARCHAR(255) NOT NULL, username VARCHAR(100) UNIQUE, phone VARCHAR(50) UNIQUE, email VARCHAR(255) UNIQUE, password_hash VARCHAR(255) NOT NULL, role VARCHAR(50) CHECK (role IN ('admin', 'teacher', 'student')), status VARCHAR(50) DEFAULT 'pending', class_id INTEGER, created_at TIMESTAMP DEFAULT NOW() ); -- جدول الصفوف CREATE TABLE classes ( id SERIAL PRIMARY KEY, name VARCHAR(255) NOT NULL, grade_level VARCHAR(100), description TEXT, created_at TIMESTAMP DEFAULT NOW() ); -- جدول المواد CREATE TABLE subjects ( id SERIAL PRIMARY KEY, name VARCHAR(255) NOT NULL, description TEXT, created_at TIMESTAMP DEFAULT NOW() ); -- جدول ربط المواد بالصفوف CREATE TABLE class_subjects ( id SERIAL PRIMARY KEY, class_id INTEGER REFERENCES classes(id) ON DELETE CASCADE, subject_id INTEGER REFERENCES subjects(id) ON DELETE CASCADE, created_at TIMESTAMP DEFAULT NOW(), UNIQUE(class_id, subject_id) ); -- جدول تعيينات المعلمين CREATE TABLE teacher_assignments ( id SERIAL PRIMARY KEY, teacher_id INTEGER REFERENCES users(id) ON DELETE CASCADE, class_id INTEGER REFERENCES classes(id) ON DELETE CASCADE, subject_id INTEGER REFERENCES subjects(id) ON DELETE CASCADE, created_at TIMESTAMP DEFAULT NOW() ); -- جدول الامتحانات CREATE TABLE exams ( id SERIAL PRIMARY KEY, title VARCHAR(255) NOT NULL, description TEXT, class_id INTEGER REFERENCES classes(id), subject_id INTEGER REFERENCES subjects(id), teacher_id INTEGER REFERENCES users(id), duration_minutes INTEGER DEFAULT 60, passing_score INTEGER DEFAULT 50, status VARCHAR(50) DEFAULT 'draft', created_at TIMESTAMP DEFAULT NOW() ); -- جدول الأسئلة CREATE TABLE questions ( id SERIAL PRIMARY KEY, teacher_id INTEGER REFERENCES users(id), class_id INTEGER REFERENCES classes(id), subject_id INTEGER REFERENCES subjects(id), question_text TEXT, question_image TEXT, type VARCHAR(50) DEFAULT 'mcq', points INTEGER DEFAULT 5, difficulty VARCHAR(20) DEFAULT 'medium', options TEXT, correct_answer TEXT, status VARCHAR(50) DEFAULT 'pending', created_at TIMESTAMP DEFAULT NOW() );

🔐 نظام تسجيل الدخول والصلاحيات

الدورطريقة الدخولالصلاحيات
أدمناسم مستخدم + كلمة سريرى كل شيء - صلاحيات كاملة
معلماسم مستخدم + كلمة سريرى فقط ما يخصه (صفوفه، مواده، طلابه)
طالبرقم هاتف + كلمة سريرى فقط الامتحانات المنشورة في صفه

🔧 الدوال الرئيسية في supabase-config.js

// === المصادقة === loginUser(identifier, password) registerStudent(name, phone, password, classId, email) hashPassword(password) // === المستخدمين === fetchUsers(filters) createUser(userData) updateUserStatus(id, status) deleteUser(id) resetUserPassword(id, newPassword) // === الصفوف والمواد === fetchClasses() createClass(data) updateClass(id, data) deleteClass(id) fetchSubjects() createSubject(data) fetchClassSubjects(classId) addSubjectToClass(classId, subjectId) // === تعيينات المعلمين === fetchTeacherAssignments(teacherId) addTeacherAssignment(teacherId, classId, subjectId) removeTeacherAssignment(id) // === الامتحانات === fetchExams(filters) createExam(examData) updateExamStatus(id, status) deleteExam(id) // === الأسئلة === fetchQuestions(filters) createQuestion(questionData) updateQuestionStatus(id, status, feedback) deleteQuestion(id) // === المجموعات === fetchGroups(filters) createGroup(groupData) updateGroup(id, data) deleteGroup(id) joinGroup(groupId, studentId) updateGroupMemberStatus(groupId, studentId, status) fetchGroupMembers(groupId) // === طلبات الامتحان === fetchExamRequests(filters) createExamRequest(studentId, examId) updateExamRequestStatus(id, status) // === محاولات الامتحان === fetchExamAttempts(filters) startExamAttempt(examId, studentId) saveAnswer(attemptId, questionId, answer) submitExamAttempt(attemptId, answers, score, totalPoints) // === الإحصائيات === fetchAdminStats() fetchTeacherStats(teacherId) fetchStudentStats(studentId)
